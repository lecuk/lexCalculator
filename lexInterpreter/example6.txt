// Program which searches for prime numbers!

// Modify maxNumber to search for more numbers
maxNumber = 50

numberToCheck = 1
~while (numberToCheck <= maxNumber)
{
	maxDivisor = numberToCheck / 2
	zeroRemainderDivisor = 1
	numberToDivideBy = 2
	
	~while (numberToDivideBy <= maxDivisor)
	{
		~if (numberToCheck % numberToDivideBy == 0)
		{
			zeroRemainderDivisor = numberToDivideBy
			~output numberToCheck
			~output " is divisible by "
			~output zeroRemainderDivisor
			~newline
		}

		numberToDivideBy += 1
	}	
	
	~if (zeroRemainderDivisor == 1)
	{
		~output numberToCheck
		~output " is PRIME"
		~newline
	}
	~newline

	numberToCheck += 1
}
