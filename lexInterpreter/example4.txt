f1(x) = x^2
f2(x, y) = x^2 / ln(y)

// !!!!!!! important !!!!!!!!
// this is special format of finding a derivative: d:x
// firstly you write "d:" and after you write name of argument being differentiated and then function with parameters

df1(x) = d:x f1(x)

// example for multiple arguments
df2(x, y) = d:y f2(y, x)

t(x, y, z) = (x == y ?? x == y - 1) ? z : 0

~output t(1, 2, 3)
~newline
~output t(3, 3, 4)
~newline
